1root@6f2c0dc55d26:/app# ip a s eth0
2106: eth0@if107: <BROADCAST,MULTICAST,UP,LOWER_UP> mtu 1500 qdisc noqueue state UP group default
3    link/ether 02:42:ac:11:00:03 brd ff:ff:ff:ff:ff:ff link-netnsid 0
4    inet 172.17.0.3/16 brd 172.17.255.255 scope global eth0
5       valid_lft forever preferred_lft forever

**eth0@if107**本质是一个 veth 虚拟网卡的容器侧端口，与宿主机上的一个 veth 接口成对存在。Docker 默认 bridge 网络模式（通过 docker0 网桥连接容器与宿主机）

字段 说明 解释
106 接口索引号（interface index），内核为该网络设备分配的唯一整数 ID，用于内部管理（如 ip link set dev 106 down 可用，但通常用名字更安全）。
eth0 该容器内看到的网卡名称，是容器命名空间（network namespace）中的名字
@if107 表示这是一个 veth（virtual ethernet）对的一端，@if107 指向对端接口在宿主机网络命名空间中的索引号是 107（即宿主机上有一个名为 vethxxxxxx@if106 的接口，其对端索引为 106 —— 这里数字对应关系是“互指”的） eth0@if107 表示：
“我在容器里叫 eth0，我的另一头（宿主机侧）的网卡接口在宿主机上编号是 107”。
这是一个 veth-pair（虚拟以太网对） 的容器侧视图。
BROADCAST 支持广播（如 IPv4 ARP、DHCP）
MULTICAST 支持组播（如 mDNS、IGMP）
UP 接口已启用（ip link set eth0 up 后置位）
LOWER_UP 物理链路层已就绪（veth 对端已连接且 UP）→ 表示 veth pair 已正确建立并激活 注意：LOWER_UP 是 veth 的关键健康指标 —— 若缺失，说明容器与宿主机网络未连通
mtu 1500 Maximum Transmission Unit：最大传输单元，即该接口能发送/接收的最大 IP 数据包大小（不含以太网头） 1500 是以太网标准值，Docker 默认 bridge 网络使用此值。
qdisc noqueue qdisc = queueing discipline（排队规则），即 Linux 流量控制（TC）的队列机制。 veth 设备的默认行为 noqueue 表示：该接口不启用排队（无缓冲队列），数据包直接发送（适用于 veth 这类低延迟、高吞吐的虚拟设备）。
state UP 接口的操作状态（operstate），表示该接口当前已启用且链路就绪 其他常见值：DOWN（禁用）、NO-CARRIER（物理断开，veth 中极少出现）、LOWERLAYERDOWN（对端 down）等
group default 接口所属的 link group（分组），用于批量操作（如 ip link set group default down）
link/ether 02:42:ac:11:00:03 接口的 MAC 地址，02:42:ac:11:00:03 是 Docker 自动生成的 MAC（格式：02:42:<网络前缀>） ac:11:00:03 → 十六进制转十进制：172.17.0.3 → 正好对应下面的 IPv4 地址
brd ff:ff:ff:ff:ff:ff Broadcast MAC address：以太网广播地址（全 F），所有设备都会接收目的为此地址的帧。
link-netnsid 0 表示该接口的 对端（peer）位于宿主机的网络命名空间中，且该命名空间在内核中被标记为 ID 0。Docker 使用 netnsid（network namespace ID）来快速定位对端命名空间。 0 通常代表 初始/宿主机网络命名空间（即 host netns）。 这是确认“该容器使用 host 网络之外的独立网络栈（bridge 模式）”的关键证据。
inet 172.17.0.3/16 容器的 IPv4 地址 + 子网掩码（即 255.255.0.0），属于 Docker 默认网桥 docker0 的网段 172.17.0.0/16
brd 172.17.255.255 该子网的广播地址（172.17.0.0/16 的广播地址 = 网络地址 + 全 1 主机位）
scope global 该地址是全局有效的（非链路本地 link 或主机本地 host），可被外部路由（在 Docker bridge 模式下，需通过 docker0 网桥转发）
valid_lft forever 地址有效生命周期（过期后不再使用） forever：表示该地址是静态配置、永不失效（符合 Docker 容器 IP 分配行为）。
preferred_lft forever 首选生命周期（过期后不再作为新连接的源地址，但旧连接仍可用） forever：表示该地址是静态配置、永不失效（符合 Docker 容器 IP 分配行为）。

 1# 按状态过滤
 2$ ss -t state established
 3$ ss -t state listening

 5$ ss -t state fin-wait-1
 6
 7# 常用状态：
 8# established    已建立连接

四、tc 命令 - 流量控制

1. 基本概念

1# 查看接口的流量控制配置
2$ tc qdisc show dev eth0
3$ tc class show dev eth0
4$ tc filter show dev eth0
5
6# 查看统计信息
7$ tc -s qdisc show dev eth0
8$ tc -s class show dev eth0

2. 队列规则（QDisc）

 1# 添加 pfifo_fast（默认）
 2$ sudo tc qdisc add dev eth0 root pfifo_fast
 3
 4# 添加 HTB（分层令牌桶）
 5$ sudo tc qdisc add dev eth0 root handle 1: htb default 30
 6
 7# 添加 TBF（令牌桶过滤器）
 8$ sudo tc qdisc add dev eth0 root tbf rate 1mbit burst 32kbit latency 400ms
 9
10# 添加 SFQ（随机公平队列）
11$ sudo tc qdisc add dev eth0 root sfq perturb 10
12
13# 删除队列规则
14$ sudo tc qdisc del dev eth0 root

3. 分类（Class）

1# 创建 HTB 类
2$ sudo tc class add dev eth0 parent 1: classid 1:1 htb rate 10mbit ceil 10mbit
3$ sudo tc class add dev eth0 parent 1:1 classid 1:10 htb rate 5mbit ceil 10mbit
4$ sudo tc class add dev eth0 parent 1:1 classid 1:20 htb rate 3mbit ceil 5mbit
5
6# 查看类
7$ tc class show dev eth0

4. 过滤器（Filter）

1# 基于端口的过滤器
2$ sudo tc filter add dev eth0 protocol ip parent 1:0 prio 1 u32 match ip dport 80 0xffff flowid 1:10
3$ sudo tc filter add dev eth0 protocol ip parent 1:0 prio 2 u32 match ip dport 443 0xffff flowid 1:20
4
5# 基于 IP 的过滤器
6$ sudo tc filter add dev eth0 protocol ip parent 1:0 prio 3 u32 match ip src 192.168.1.100 flowid 1:30
7
8# 删除过滤器
9$ sudo tc filter del dev eth0 parent 1:0 protocol ip prio 1

 1#!/bin/bash
 2# network-qos-setup.sh
 3
 4INTERFACE="eth0"
 5UPLOAD_LIMIT="10mbit"    # 上传限制 10Mbps
 6DOWNLOAD_LIMIT="50mbit"  # 下载限制 50Mbps
 7

 9

12sudo tc qdisc del dev "$INTERFACE" ingress 2>/dev/null
13

15echo "配置上传限速: $UPLOAD_LIMIT"

17
18# 创建根类
19sudo tc class add dev "$INTERFACE" parent 1: classid 1:1 htb rate "$UPLOAD_LIMIT" ceil "$UPLOAD_LIMIT"
20

22sudo tc class add dev "$INTERFACE" parent 1:1 classid 1:10 htb rate 2mbit ceil "$UPLOAD_LIMIT" prio 1  # 高优先级
23sudo tc class add dev "$INTERFACE" parent 1:1 classid 1:20 htb rate 5mbit ceil "$UPLOAD_LIMIT" prio 2  # 中优先级
24sudo tc class add dev "$INTERFACE" parent 1:1 classid 1:30 htb rate 3mbit ceil "$UPLOAD_LIMIT" prio 3  # 低优先级
25
26# 为每个类添加队列规则
27sudo tc qdisc add dev "$INTERFACE" parent 1:10 handle 10: sfq perturb 10
28sudo tc qdisc add dev "$INTERFACE" parent 1:20 handle 20: sfq perturb 10
29sudo tc qdisc add dev "$INTERFACE" parent 1:30 handle 30: sfq perturb 10
30
31# 3. 配置过滤器
32echo "配置流量分类:"
33
34# SSH 流量 - 高优先级
35sudo tc filter add dev "$INTERFACE" protocol ip parent 1:0 prio 1 u32 \
36    match ip dport 22 0xffff flowid 1:10
37
38# HTTP/HTTPS 流量 - 中优先级
39sudo tc filter add dev "$INTERFACE" protocol ip parent 1:0 prio 2 u32 \
40    match ip dport 80 0xffff flowid 1:20

42    match ip dport 443 0xffff flowid 1:20
43

47

57sudo tc filter add dev "$INTERFACE" parent ffff: protocol ip prio 1 u32 \
58    match u32 0 0 police rate "$DOWNLOAD_LIMIT" burst 100k drop flowid :1 \
59    action mirred egress redirect dev ifb0
60

5. 网络故障自动恢复

 1#!/bin/bash
 2# network-auto-recover.sh
 3
 4INTERFACE="eth0"
 5GATEWAY="192.168.1.1"
 6CHECK_INTERVAL=60  # 检查间隔（秒）
 7LOG_FILE="/var/log/network-recover.log"
 8
 9log() {
10    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" | tee -a "$LOG_FILE"
11}
12
13check_network() {
14    # 检查接口状态
15    if ! ip link show "$INTERFACE" | grep -q "state UP"; then
16        log "接口 $INTERFACE 未启用，尝试启用..."
17        sudo ip link set "$INTERFACE" up
18        sleep 2
19    fi
20    
21    # 检查网关连通性
22    if ping -c 2 -W 1 "$GATEWAY" > /dev/null 2>&1; then
23        return 0
24    else
25        log "网关 $GATEWAY 不可达"
26        return 1
27    fi
28}
29
30recover_network() {
31    log "开始网络恢复流程..."
32    
33    # 1. 重启网络接口
34    log "重启接口 $INTERFACE..."
35    sudo ip link set "$INTERFACE" down
36    sleep 2
37    sudo ip link set "$INTERFACE" up
38    sleep 5
39    
40    # 2. 刷新 DHCP（如果使用）
41    if pgrep dhclient > /dev/null; then
42        log "重启 DHCP 客户端..."
43        sudo pkill dhclient
44        sudo dhclient "$INTERFACE"
45    fi
46    
47    # 3. 刷新 ARP 缓存
48    log "刷新 ARP 缓存..."
49    sudo ip neigh flush dev "$INTERFACE"
50    
51    # 4. 刷新路由缓存
52    log "刷新路由缓存..."
53    sudo ip route flush cache
54    
55    # 5. 重启 DNS
56    log "重启 DNS 服务..."
57    sudo systemctl restart systemd-resolved 2>/dev/null || true
58    
59    log "网络恢复完成"
60}
61
62# 主循环
63log "启动网络监控，检查间隔: ${CHECK_INTERVAL}秒"
64log "监控接口: $INTERFACE, 网关: $GATEWAY"
65
66while true; do
67    if ! check_network; then
68        log "网络连接异常，尝试恢复..."
69        recover_network
70        
71        # 恢复后再次检查
72        sleep 10
73        if check_network; then
74            log "网络恢复成功"
75        else
76            log "网络恢复失败，等待下一次检查"
77        fi
78    else
79        log "网络连接正常"
80    fi
81    
82    sleep "$CHECK_INTERVAL"
83done

 19    echo "时间戳,接口,RX包数,RX字节,TX包数,TX字节,RX错误,TX错误,RX丢弃,TX丢弃" > "$OUTPUT_DIR/traffic-stats.csv"
 20    START_TIME=$(date +%s)
 21    
 22    for i in $(seq 1 $DURATION); do
 23        CURRENT_TIME=$(date +%s)
 24        ELAPSED=$((CURRENT_TIME - START_TIME))
 25        
 26        # 获取详细的接口统计
 27        STATS=$(ip -s -s link show "$INTERFACE" 2>/dev/null | grep -A 3 "RX\|TX")
 28        
 29        if [ -n "$STATS" ]; then
 30            RX_LINE1=$(echo "$STATS" | grep -A 1 "RX:" | tail -1)
 31            RX_LINE2=$(echo "$STATS" | grep -A 2 "RX:" | tail -1)
 32            TX_LINE1=$(echo "$STATS" | grep -A 1 "TX:" | tail -1)
 33            TX_LINE2=$(echo "$STATS" | grep -A 2 "TX:" | tail -1)
 34            
 35            RX_PKTS=$(echo "$RX_LINE1" | awk '{print \$1}')
 36            RX_BYTES=$(echo "$RX_LINE1" | awk '{print \$2}')
 37            RX_ERRORS=$(echo "$RX_LINE2" | awk '{print \$1}')
 38            RX_DROPPED=$(echo "$RX_LINE2" | awk '{print \$2}')
 39            
 40            TX_PKTS=$(echo "$TX_LINE1" | awk '{print \$1}')
 41            TX_BYTES=$(echo "$TX_LINE1" | awk '{print \$2}')
 42            TX_ERRORS=$(echo "$TX_LINE2" | awk '{print \$1}')
 43            TX_DROPPED=$(echo "$TX_LINE2" | awk '{print \$2}')
 44            
 45            echo "${ELAPSED},${INTERFACE},${RX_PKTS},${RX_BYTES},${TX_PKTS},${TX_BYTES},${RX_ERRORS},${TX_ERRORS},${RX_DROPPED},${TX_DROPPED}" \
 46                >> "$OUTPUT_DIR/traffic-stats.csv"
 47        fi
 48        
 49        sleep 1
 50    done
 51) &
 52STATS_PID=$!

 54# 3. 连接状态监控
 55echo "3. 监控连接状态..."
 56(

 58    START_TIME=$(date +%s)

 60    for i in $(seq 1 $DURATION); do

 62        ELAPSED=$((CURRENT_TIME - START_TIME))
 63

 65        SS_OUTPUT=$(ss -s 2>/dev/null)
 66

 69            TCP_ESTABLISHED=$(echo "$SS_OUTPUT" | grep "ESTAB" | awk '{print \$2}' | tr -d ',')
 70            UDP_TOTAL=$(echo "$SS_OUTPUT" | grep "UDP:" | awk '{print \$2}' | tr -d ',')

 77        sleep 5  # 每5秒采样一次
 78    done

 80CONN_PID=$!

 82# 4. 路由统计监控
 83echo "4. 监控路由统计..."
 84(
 85    echo "时间戳,目标网络,出站字节,出站包数,入站字节,入站包数" > "$OUTPUT_DIR/route-stats.csv"

 92        CURRENT_TIME=$(date +%s)
 93        ELAPSED=$((CURRENT_TIME - START_TIME))

 95        # 获取路由统计
 96        RTACCT_OUTPUT=$(rtacct 2>/dev/null | tail -n +2)
 97        
 98        while read -r line; do
 99            if [ -n "$line" ]; then
100                DEST=$(echo "$line" | awk '{print \$1}')
101                BYTES_OUT=$(echo "$line" | awk '{print \$4}')
102                PKTS_OUT=$(echo "$line" | awk '{print \$5}')
103                BYTES_IN=$(echo "$line" | awk '{print \$8}')
104                PKTS_IN=$(echo "$line" | awk '{print \$9}')
105                
106                echo "${ELAPSED},${DEST},${BYTES_OUT},${PKTS_OUT},${BYTES_IN},${PKTS_IN}" \
107                    >> "$OUTPUT_DIR/route-stats.csv"

109        done <<< "$RTACCT_OUTPUT"
110        
111        sleep 10  # 每10秒采样一次
112    done

116# 5. 网络错误监控
117echo "5. 监控网络错误..."
118(
119    echo "时间戳,接口,接收错误,发送错误,接收丢弃,发送丢弃,接收FIFO错误,发送FIFO错误,接收帧错误,发送碰撞" > "$OUTPUT_DIR/error-stats.csv"
120    START_TIME=$(date +%s)
121    
122    for i in $(seq 1 $DURATION); do
123        CURRENT_TIME=$(date +%s)
124        ELAPSED=$((CURRENT_TIME - START_TIME))
125        
126        # 从 /proc/net/dev 获取详细统计
127        PROC_STATS=$(grep "$INTERFACE:" /proc/net/dev)
128        
129        if [ -n "$PROC_STATS" ]; then
130            # 解析 /proc/net/dev 格式
131            # 字段顺序: bytes packets errs drop fifo frame compressed multicast|bytes packets errs drop fifo colls carrier compressed
132            RX_BYTES=$(echo "$PROC_STATS" | awk '{print \$2}')
133            RX_PKTS=$(echo "$PROC_STATS" | awk '{print \$3}')
134            RX_ERRS=$(echo "$PROC_STATS" | awk '{print \$4}')
135            RX_DROP=$(echo "$PROC_STATS" | awk '{print \$5}')
136            RX_FIFO=$(echo "$PROC_STATS" | awk '{print \$6}')
137            RX_FRAME=$(echo "$PROC_STATS" | awk '{print \$7}')
138            RX_COMPRESSED=$(echo "$PROC_STATS" | awk '{print \$8}')
139            RX_MULTICAST=$(echo "$PROC_STATS" | awk '{print \$9}')
140            
141            TX_BYTES=$(echo "$PROC_STATS" | awk '{print \$10}')
142            TX_PKTS=$(echo "$PROC_STATS" | awk '{print \$11}')
143            TX_ERRS=$(echo "$PROC_STATS" | awk '{print \$12}')
144            TX_DROP=$(echo "$PROC_STATS" | awk '{print \$13}')
145            TX_FIFO=$(echo "$PROC_STATS" | awk '{print \$14}')
146            TX_COLLS=$(echo "$PROC_STATS" | awk '{print \$15}')
147            TX_CARRIER=$(echo "$PROC_STATS" | awk '{print \$16}')
148            TX_COMPRESSED=$(echo "$PROC_STATS" | awk '{print \$17}')
149            
150            echo "${ELAPSED},${INTERFACE},${RX_ERRS},${TX_ERRS},${RX_DROP},${TX_DROP},${RX_FIFO},${TX_FIFO},${RX_FRAME},${TX_COLLS}" \
151                >> "$OUTPUT_DIR/error-stats.csv"
152        fi
153        
154        sleep 1
155    done
156) &
157ERROR_PID=$!

159# 6. 带宽使用监控
160echo "6. 监控带宽使用..."
161(
162    echo "时间戳,接口,接收速率(KB/s),发送速率(KB/s),总速率(KB/s)" > "$OUTPUT_DIR/bandwidth-usage.csv"
163    START_TIME=$(date +%s)
164    
165    # 获取初始统计
166    PREV_STATS=$(grep "$INTERFACE:" /proc/net/dev)
167    PREV_RX_BYTES=$(echo "$PREV_STATS" | awk '{print \$2}')
168    PREV_TX_BYTES=$(echo "$PREV_STATS" | awk '{print \$10}')
169    PREV_TIME=$START_TIME
170    
171    for i in $(seq 1 $DURATION); do
172        sleep 1
173        
174        CURRENT_TIME=$(date +%s)
175        ELAPSED=$((CURRENT_TIME - START_TIME))
176        
177        # 获取当前统计
178        CURR_STATS=$(grep "$INTERFACE:" /proc/net/dev)
179        CURR_RX_BYTES=$(echo "$CURR_STATS" | awk '{print \$2}')
180        CURR_TX_BYTES=$(echo "$CURR_STATS" | awk '{print \$10}')
181        
182        # 计算速率（字节/秒）
183        TIME_DELTA=$((CURRENT_TIME - PREV_TIME))
184        if [ $TIME_DELTA -gt 0 ]; then
185            RX_RATE=$(((CURR_RX_BYTES - PREV_RX_BYTES) / TIME_DELTA))
186            TX_RATE=$(((CURR_TX_BYTES - PREV_TX_BYTES) / TIME_DELTA))
187            TOTAL_RATE=$((RX_RATE + TX_RATE))
188            
189            # 转换为 KB/s
190            RX_RATE_KB=$((RX_RATE / 1024))
191            TX_RATE_KB=$((TX_RATE / 1024))
192            TOTAL_RATE_KB=$((TOTAL_RATE / 1024))
193            
194            echo "${ELAPSED},${INTERFACE},${RX_RATE_KB},${TX_RATE_KB},${TOTAL_RATE_KB}" \
195                >> "$OUTPUT_DIR/bandwidth-usage.csv"
196        fi
197        
198        # 更新前一时刻的值
199        PREV_RX_BYTES=$CURR_RX_BYTES
200        PREV_TX_BYTES=$CURR_TX_BYTES
201        PREV_TIME=$CURRENT_TIME
202    done
203) &

234        done
