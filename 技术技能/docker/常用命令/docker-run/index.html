Docker run 命令详解

📚 命令概述

docker run 是 Docker 中最核心、最常用的命令，用于从镜像创建并启动一个新的容器。

基本语法：

1docker run [OPTIONS] IMAGE [COMMAND] [ARG...]

🎯 基本使用示例

1. 最简单的运行

1# 运行一个容器，执行命令后退出
2docker run ubuntu echo "Hello Docker"
3
4# 运行并进入交互式终端
5docker run -it ubuntu bash
6
7# 运行后台容器
8docker run -d nginx

2. 常用选项分类

🔧 容器运行模式选项

交互式 vs 后台运行

 1# 交互式运行（-i -t 通常一起使用）
 2docker run -it ubuntu bash
 3# -i, --interactive: 保持 STDIN 打开
 4# -t, --tty: 分配一个伪终端
 5
 6# 后台运行（守护进程模式）
 7docker run -d nginx
 8# -d, --detach: 后台运行容器
 9
10# 后台运行但附加日志
11docker run -d --log-driver=json-file nginx

容器命名和管理

 1# 指定容器名称
 2docker run --name my-container nginx
 3
 4# 自动删除容器（退出时）
 5docker run --rm ubuntu echo "临时任务"
 6
 7# 重启策略
 8docker run --restart=always nginx
 9# no: 不自动重启（默认）
10# on-failure[:max-retries]: 失败时重启
11# always: 总是重启
12# unless-stopped: 除非手动停止，否则总是重启

🌐 网络配置选项

网络模式

 1# 使用主机网络（共享主机网络栈）
 2docker run --network=host nginx
 3
 4# 使用桥接网络（默认）
 5docker run --network=bridge nginx
 6
 7# 使用自定义网络
 8docker network create mynet
 9docker run --network=mynet nginx
10
11# 不联网
12docker run --network=none alpine
13
14# 容器网络（连接到其他容器）
15docker run --network=container:existing-container nginx

端口映射

 1# 简单端口映射
 2docker run -p 8080:80 nginx
 3# 主机端口:容器端口
 4
 5# 多个端口映射
 6docker run -p 8080:80 -p 8443:443 nginx
 7
 8# 指定主机IP
 9docker run -p 127.0.0.1:8080:80 nginx
10
11# 随机主机端口
12docker run -p 80 nginx
13docker port <container_id>  # 查看分配的端口
14
15# 使用 UDP 端口
16docker run -p 53:53/udp dns-server

DNS 和主机名

1# 设置主机名
2docker run --hostname myhost ubuntu
3
4# 添加 hosts 条目
5docker run --add-host db.local:192.168.1.100 ubuntu
6
7# 自定义 DNS
8docker run --dns 8.8.8.8 ubuntu
9docker run --dns-search example.com ubuntu

💾 存储和卷选项

数据卷（Volumes）

 1# 创建匿名卷
 2docker run -v /data ubuntu
 3
 4# 挂载主机目录
 5docker run -v /host/path:/container/path ubuntu
 6
 7# 挂载数据卷
 8docker volume create mydata
 9docker run -v mydata:/data ubuntu
10
11# 只读挂载
12docker run -v /host/path:/container/path:ro ubuntu
13
14# 挂载多个目录
15docker run -v /host/data:/data -v /host/config:/config ubuntu

tmpfs 挂载

1# 内存文件系统
2docker run --tmpfs /tmp:size=100M,mode=1777 ubuntu
3
4# 多个 tmpfs
5docker run --tmpfs /tmp --tmpfs /run ubuntu

绑定挂载高级选项

 1# 使用 --mount（更详细的选项）
 2docker run \
 3  --mount type=bind,source=/host/path,target=/container/path,readonly \
 4  ubuntu
 5
 6# 挂载类型：
 7# type=bind: 绑定挂载
 8# type=volume: 数据卷
 9# type=tmpfs: 临时文件系统
10# type=pipe: 命名管道（Windows）
11
12# 传播选项
13docker run \
14  --mount type=bind,source=/host,target=/container,bind-propagation=rshared \
15  ubuntu

⚙️ 资源限制选项

CPU 限制

 1# 限制 CPU 使用（相对权重）
 2docker run --cpu-shares=512 ubuntu
 3# 默认 1024，值越高优先级越高
 4
 5# 限制 CPU 核心数
 6docker run --cpus=1.5 ubuntu
 7docker run --cpus=0.5 ubuntu
 8
 9# 绑定到特定 CPU 核心
10docker run --cpuset-cpus="0,3" ubuntu
11docker run --cpuset-cpus="0-2" ubuntu

内存限制

 1# 内存限制
 2docker run -m 512m ubuntu
 3# 或
 4docker run --memory=512m ubuntu
 5
 6# 内存+交换分区限制
 7docker run -m 512m --memory-swap=1g ubuntu
 8# 如果只设置 memory，swap = memory * 2
 9# 如果设置为 -1，不限制 swap
10
11# 内存预留（软限制）
12docker run --memory-reservation=256m ubuntu
13
14# OOM 优先级
15docker run --oom-kill-disable ubuntu
16docker run --oom-score-adj=-500 ubuntu

其他资源限制

 1# 设备限制
 2docker run --device=/dev/sda:/dev/xvda ubuntu
 3docker run --device-read-bps=/dev/sda:1mb ubuntu
 4docker run --device-write-iops=/dev/sda:100 ubuntu
 5
 6# 进程数限制
 7docker run --pids-limit=100 ubuntu
 8
 9# 内核参数
10docker run --ulimit nofile=1024:1024 ubuntu
11# ulimit 选项：
12# core: 核心文件大小
13# cpu: CPU时间（秒）
14# data: 数据段大小
15# fsize: 文件大小
16# locks: 文件锁数
17# memlock: 锁定内存
18# msgqueue: 消息队列大小
19# nice: 优先级
20# nofile: 打开文件数
21# nproc: 进程数
22# rss: 常驻集大小
23# rtprio: 实时优先级
24# sigpending: 挂起信号
25# stack: 栈大小

🔐 安全选项

用户和权限

 1# 指定运行用户
 2docker run -u 1000 ubuntu
 3docker run --user=appuser ubuntu
 4
 5# 用户命名空间
 6docker run --userns=host ubuntu
 7
 8# 安全选项（SELinux/AppArmor）
 9docker run --security-opt label=type:svirt_apache ubuntu
10docker run --security-opt apparmor=docker-default ubuntu
11docker run --security-opt seccomp=unconfined ubuntu
12
13# 不推荐：特权模式（获得所有权限）
14docker run --privileged ubuntu
15
16# 添加/删除能力
17docker run --cap-add=NET_ADMIN ubuntu
18docker run --cap-drop=ALL --cap-add=NET_BIND_SERVICE ubuntu

只读文件系统

1# 只读根文件系统
2docker run --read-only ubuntu
3
4# 配合 tmpfs 使用
5docker run --read-only --tmpfs /tmp ubuntu
6
7# 只读特定目录
8docker run -v /data:ro ubuntu

📦 镜像相关选项

镜像标签和拉取策略

 1# 指定镜像标签
 2docker run ubuntu:20.04
 3docker run nginx:alpine
 4
 5# 拉取策略
 6docker run --pull=always nginx  # 总是拉取最新
 7docker run --pull=missing nginx # 本地没有才拉取
 8docker run --pull=never nginx   # 只用本地镜像
 9
10# 平台架构
11docker run --platform=linux/amd64 ubuntu
12docker run --platform=linux/arm64 ubuntu

📝 环境变量

 1# 设置环境变量
 2docker run -e MYVAR=value ubuntu
 3docker run -e MYVAR ubuntu  # 从主机继承
 4
 5# 多个环境变量
 6docker run -e VAR1=value1 -e VAR2=value2 ubuntu
 7
 8# 从文件读取
 9docker run --env-file .env ubuntu
10
11# 特殊环境变量
12docker run -e "TERM=xterm-256color" ubuntu
13docker run -e "PATH=/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin" ubuntu

🎨 输出和日志

 1# 输出附加
 2docker run -a stdin -a stdout -a stderr ubuntu
 3
 4# 日志驱动
 5docker run --log-driver=json-file ubuntu
 6docker run --log-driver=syslog ubuntu
 7docker run --log-driver=journald ubuntu
 8docker run --log-driver=none ubuntu
 9
10# 日志选项
11docker run --log-opt max-size=10m --log-opt max-file=3 ubuntu
12docker run --log-opt tag="{{.ImageName}}/{{.Name}}/{{.ID}}" ubuntu

4
5# USB 设备
6docker run --device=/dev/bus/usb/001/002 ubuntu
7
8# 字符设备
9docker run --device=/dev/ttyUSB0 ubuntu

IPC 和共享内存

📊 实际应用示例

示例1：Web 应用容器

 1docker run -d \
 2  --name=my-webapp \
 3  --hostname=webapp \
 4  -p 8080:80 \
 5  -p 8443:443 \
 6  -v /data/webapp:/var/www/html \
 7  -v /data/config:/etc/nginx/conf.d \
 8  -e "NGINX_HOST=example.com" \
 9  -e "NGINX_PORT=80" \
10  --memory=512m \
11  --cpus=1 \
12  --restart=unless-stopped \
13  --log-opt max-size=10m \
14  --log-opt max-file=3 \
15  nginx:alpine

示例2：数据库容器

 1docker run -d \
 2  --name=mysql-db \
 3  -p 3306:3306 \
 4  -v mysql_data:/var/lib/mysql \
 5  -v /backup/mysql:/backup \
 6  -e MYSQL_ROOT_PASSWORD=secret \
 7  -e MYSQL_DATABASE=appdb \
 8  -e MYSQL_USER=appuser \
 9  -e MYSQL_PASSWORD=apppass \
10  --network=app-network \
11  --memory=1g \
12  --memory-swap=2g \
13  --cpus=2 \
14  --ulimit nofile=65536:65536 \
15  mysql:8.0 \
16  --character-set-server=utf8mb4 \
17  --collation-server=utf8mb4_unicode_ci

示例3：开发环境容器

 1docker run -it \
 2  --name=dev-env \
 3  -v $(pwd):/workspace \
 4  -v ~/.ssh:/root/.ssh \
 5  -v ~/.gitconfig:/root/.gitconfig \
 6  -p 3000:3000 \
 7  -p 9229:9229 \
 8  --add-host=host.docker.internal:host-gateway \
 9  -e "NODE_ENV=development" \
10  -e "DEBUG=*" \
11  --security-opt seccomp=unconfined \
12  --cap-add=SYS_PTRACE \
13  node:18-alpine \
14  sh

2  -v $(pwd):/app \

🎓 最佳实践

  1. 始终指定版本标签

    1# 好
2docker run nginx:1.21-alpine
3
4# 不好
5docker run nginx

  2. 使用命名容器便于管理

    1docker run --name my-app nginx

  3. 合理设置资源限制

    1docker run -m 512m --cpus=1 nginx

  4. 使用数据卷持久化数据

    1docker run -v app-data:/data myapp

  5. 设置合适的重启策略

    1docker run --restart=unless-stopped myapp

  6. 记录日志并设置轮转

    1docker run --log-opt max-size=10m --log-opt max-file=3 myapp

  7. 使用环境变量配置

    1docker run -e "DB_HOST=db" -e "DB_PORT=5432" myapp

  8. 清理临时容器

    1docker run --rm alpine echo "done"

🚨 常见错误和解决

 1# 错误：端口已被占用
 2docker run -p 80:80 nginx
 3# 解决：使用不同端口或停止占用端口的容器
 4
 5# 错误：容器名已存在
 6docker run --name myapp nginx
 7# 解决：删除旧容器或使用不同名称
 8
 9# 错误：权限不足
10docker run -v /etc:/container-etc ubuntu
11# 解决：使用适当权限或调整 SELinux/AppArmor
12
13# 错误：内存不足
14docker run -m 8g myapp
15# 解决：检查主机可用内存，调整限制
16
17# 错误：镜像不存在
18docker run nonexistent:latest
19# 解决：先拉取镜像 docker pull nonexistent:latest

🔗 容器连接和通信

1. 容器间链接（传统方式）

 1# 先启动数据库容器
 2docker run -d \
 3  --name db \
 4  -e MYSQL_ROOT_PASSWORD=secret \
 5  mysql:8.0
 6
 7# 启动应用容器并链接到数据库
 8docker run -d \
 9  --name app \
10  --link db:mysql \
11  -e "DB_HOST=mysql" \
12  myapp:latest

2. 使用自定义网络（推荐）

 1# 创建网络
 2docker network create app-network
 3
 4# 启动数据库容器
 5docker run -d \
 6  --name db \
 7  --network app-network \
 8  -e MYSQL_ROOT_PASSWORD=secret \
 9  mysql:8.0
10
11# 启动应用容器
12docker run -d \
13  --name app \
14  --network app-network \
15  -e "DB_HOST=db" \
16  -p 8080:80 \
17  myapp:latest

3. 容器间文件共享

 1# 使用共享卷
 2docker volume create shared-data
 3
 4# 容器1写入数据
 5docker run -it \
 6  --name writer \

5. 备份和恢复

容器数据备份

 1# 备份脚本
 2#!/bin/bash
 3# backup-container.sh
 4
 5CONTAINER=$1
 6BACKUP_DIR=${2:-/backup}
 7TIMESTAMP=$(date +%Y%m%d_%H%M%S)
 8
 9# 备份容器配置
10docker inspect $CONTAINER > $BACKUP_DIR/${CONTAINER}_${TIMESTAMP}_inspect.json
11
12# 备份数据卷
13for volume in $(docker inspect -f '{{range .Mounts}}{{if eq .Type "volume"}}{{.Name}} {{end}}{{end}}' $CONTAINER); do
14    docker run --rm \
15        -v $volume:/source \
16        -v $BACKUP_DIR:/backup \
17        alpine tar czf /backup/${volume}_${TIMESTAMP}.tar.gz -C /source .
18done
19
20# 备份绑定挂载
21for mount in $(docker inspect -f '{{range .Mounts}}{{if eq .Type "bind"}}{{.Source}} {{end}}{{end}}' $CONTAINER); do
22    tar czf $BACKUP_DIR/$(basename $mount)_${TIMESTAMP}.tar.gz -C $mount .
23done
24
25echo "备份完成: $BACKUP_DIR/${CONTAINER}_${TIMESTAMP}_*"

恢复脚本

 1#!/bin/bash
 2# restore-container.sh
 3
 4BACKUP_FILE=$1
 5CONTAINER=$2
 6
 7# 从备份恢复
 8tar xzf $BACKUP_FILE -C /tmp/backup
 9
10# 恢复数据卷
11for volume_backup in /tmp/backup/*_volume_*.tar.gz; do
12    volume_name=$(basename $volume_backup | cut -d'_' -f1)
13    docker volume create $volume_name
14    docker run --rm \
15        -v $volume_name:/target \
16        -v $(dirname $volume_backup):/backup \
17        alpine tar xzf /backup/$(basename $volume_backup) -C /target
18done
19
20# 重新创建容器（使用原始配置）
21docker run $(cat /tmp/backup/run_command.txt)
